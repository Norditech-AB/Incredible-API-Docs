import requests

    def generate_function(self, language: str, function_name: str, 
                         requirements: str, context: str = "", 
                         include_tests: bool = False) -> Dict:
        """Generate a complete function based on requirements"""
        
        prompt = f"""
        Generate a {language} function with the following specifications:
        
        Function Name: {function_name}
        Requirements: {requirements}
        Language: {language}
        Context: {context if context else 'No specific context provided'}
        Include Tests: {'Yes' if include_tests else 'No'}
        
        Please provide:
        1. A complete, production-ready function
        2. Proper error handling and input validation
        3. Clear documentation and comments
        4. Type hints (if supported by the language)
        5. Unit tests (if requested)
        6. Usage examples
        
        Follow {language} best practices and conventions.
        """
        
        response = requests.post(
            self.base_url,
            headers={
                "Content-Type": "application/json",
                "Authorization": f"Bearer {self.api_key}"
            },
            json={
                "model": "small-1",
                "stream": False,
                "messages": [{"role": "user", "content": prompt}],
                "functions": self._get_code_tools()
            }
        )
        
        return response.json()
