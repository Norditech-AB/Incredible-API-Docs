```python
import requests
import json
from typing import Dict, List, Optional
from datetime import datetime

class AIContentAgency:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = "https://api.incredible.one/v1/chat-completion"
        
    def generate_blog_post(self, topic: str, target_audience: str, 
                          tone: str = "professional", word_count: int = 1000) -> Dict:
        """Generate a complete blog post using AI"""
        
        prompt = f"""
        Create a comprehensive blog post about: {topic}
        
        Target Audience: {target_audience}
        Tone: {tone}
        Word Count: {word_count} words
        
        Please include:
        1. An engaging headline
        2. A compelling introduction
        3. Well-structured sections with subheadings
        4. Actionable insights and examples
        5. A strong conclusion with call-to-action
        6. 5-7 relevant keywords for SEO
        
        Format the response as a structured blog post with clear sections.
        """
        
        response = requests.post(
            self.base_url,
            headers={
                "Content-Type": "application/json",
                "Authorization": f"Bearer {self.api_key}"
            },
            json={
                "model": "small-1",
                "stream": False,
                "messages": [{"role": "user", "content": prompt}],
                "functions": self._get_content_tools()
            }
        )
        
        return response.json()
    
    def create_social_media_campaign(self, campaign_theme: str, 
                                   platforms: List[str], post_count: int = 5) -> Dict:
        """Generate a complete social media campaign"""
        
        platform_specifics = {
            "twitter": "280 character limit, use hashtags",
            "linkedin": "Professional tone, industry insights",
            "instagram": "Visual descriptions, engaging captions",
            "facebook": "Community-focused, shareable content"
        }
        
        prompt = f"""
        Create a social media campaign about: {campaign_theme}
        
        Platforms: {', '.join(platforms)}
        Number of posts: {post_count}
        
        For each platform, consider:
        {chr(10).join([f"- {platform}: {platform_specifics.get(platform, 'General social media')}" for platform in platforms])}
        
        Generate {post_count} unique posts that can be adapted for each platform.
        Include relevant hashtags and engagement prompts.
        """
        
        response = requests.post(
            self.base_url,
            headers={
                "Content-Type": "application/json",
                "Authorization": f"Bearer {self.api_key}"
            },
            json={
                "model": "small-1",
                "stream": False,
                "messages": [{"role": "user", "content": prompt}],
                "functions": self._get_content_tools()
            }
        )
        
        return response.json()
    
    def write_email_sequence(self, sequence_type: str, email_count: int,
                            target_audience: str, call_to_action: str) -> Dict:
        """Create an email marketing sequence"""
        
        sequence_templates = {
            "welcome": "Welcome new subscribers and introduce your brand",
            "nurture": "Build relationships and provide value over time",
            "sales": "Convert prospects into customers",
            "re-engagement": "Win back inactive subscribers"
        }
        
        prompt = f"""
        Create a {sequence_type} email sequence for: {target_audience}
        
        Number of emails: {email_count}
        Call to action: {call_to_action}
        
        Sequence purpose: {sequence_templates.get(sequence_type, sequence_type)}
        
        Each email should:
        1. Have a compelling subject line
        2. Include personalized content
        3. Build on previous emails
        4. Guide toward the call to action
        5. Be optimized for open rates and engagement
        
        Format as a structured email sequence with clear progression.
        """
        
        response = requests.post(
            self.base_url,
            headers={
                "Content-Type": "application/json",
                "Authorization": f"Bearer {self.api_key}"
            },
            json={
                "model": "small-1",
                "stream": False,
                "messages": [{"role": "user", "content": prompt}],
                "functions": self._get_content_tools()
            }
        )
        
        return response.json()
    
    def _get_content_tools(self) -> List[Dict]:
        """Get the content creation tools available"""
        return [
            {
                "name": "research_topic",
                "description": "Research current trends and information on a topic",
                "parameters": {
                    "type": "object",
                    "properties": {
                        "topic": {"type": "string"},
                        "research_depth": {"type": "string", "enum": ["basic", "comprehensive", "expert"]}
                    },
                    "required": ["topic"]
                }
            },
            {
                "name": "optimize_content",
                "description": "Optimize content for SEO and engagement",
                "parameters": {
                    "type": "object",
                    "properties": {
                        "content": {"type": "string"},
                        "optimization_type": {"type": "string", "enum": ["seo", "engagement", "conversion"]}
                    },
                    "required": ["content"]
                }
            }
        ]

# Usage example
if __name__ == "__main__":
```
