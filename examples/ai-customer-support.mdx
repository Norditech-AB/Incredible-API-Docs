```python
import requests
import json
from typing import Dict, List, Optional
from datetime import datetime
import uuid

class AICustomerSupport:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = "https://api.incredible.one/v1/chat-completion"
        
    def handle_customer_inquiry(self, customer_message: str, 
                               customer_context: Dict, 
                               product_info: Dict) -> Dict:
        """Automatically handle a customer inquiry"""
        
        # Step 1: Classify the inquiry
        classification = self._classify_inquiry(customer_message, customer_context, product_info)
        
        # Step 2: Generate appropriate response
        if classification["can_auto_resolve"]:
            response = self._generate_automated_response(
                inquiry_type=classification["type"],
                customer_context=customer_context,
                customer_message=customer_message
            )
            
            return {
                "ticket_id": str(uuid.uuid4()),
                "status": "resolved",
                "classification": classification,
                "response": response,
                "escalation_needed": False,
                "confidence_score": classification["confidence"]
            }
        else:
            # Step 3: Determine if escalation is needed
            escalation = self._should_escalate(
                conversation_history=[customer_message],
                issue_complexity=classification["complexity"]
            )
            
            return {
                "ticket_id": str(uuid.uuid4()),
                "status": "needs_human",
                "classification": classification,
                "escalation_reason": escalation["reason"],
                "priority": escalation["priority"],
                "assigned_agent": None
            }
    
    def _classify_inquiry(self, message: str, context: Dict, product: Dict) -> Dict:
        """Classify customer inquiry by type and complexity"""
        
        prompt = f"""
        Classify this customer inquiry:
        
        Message: {message}
        Customer Context: {json.dumps(context, indent=2)}
        Product Info: {json.dumps(product, indent=2)}
        
        Please classify:
        1. Inquiry type (technical, billing, product, general)
        2. Complexity level (simple, moderate, complex)
        3. Whether it can be auto-resolved
        4. Confidence score (0-100)
        5. Suggested response category
        
        Return as structured classification data.
        """
        
        response = requests.post(
            self.base_url,
            headers={
                "Content-Type": "application/json",
                "Authorization": f"Bearer {self.api_key}"
            },
            json={
                "model": "small-1",
                "stream": False,
                "messages": [{"role": "user", "content": prompt}],
                "functions": self._get_support_tools()
            }
        )
        
        return response.json()
    
    def _generate_automated_response(self, inquiry_type: str, 
                                   customer_context: Dict, 
                                   customer_message: str) -> Dict:
        """Generate automated customer support response"""
        
        prompt = f"""
        Generate a helpful customer support response for:
        
        Inquiry Type: {inquiry_type}
        Customer Message: {customer_message}
        Customer Context: {json.dumps(customer_context, indent=2)}
        
        The response should:
        1. Be personalized and friendly
        2. Address the specific issue
        3. Provide clear solutions or next steps
        4. Include relevant product information
        5. End with a helpful closing
        
        Format as a professional customer support response.
        """
        
        response = requests.post(
            self.base_url,
            headers={
                "Content-Type": "application/json",
                "Authorization": f"Bearer {self.api_key}"
            },
            json={
                "model": "small-1",
                "stream": False,
                "messages": [{"role": "user", "content": prompt}],
                "functions": self._get_support_tools()
            }
        )
        
        return response.json()
    
    def _should_escalate(self, conversation_history: List[str], 
                         issue_complexity: str) -> Dict:
        """Determine if ticket should be escalated to human agent"""
        
        prompt = f"""
        Analyze this support conversation and determine if escalation is needed:
        
        Conversation History: {json.dumps(conversation_history, indent=2)}
        Issue Complexity: {issue_complexity}
        
        Consider:
        1. Has the AI attempted multiple solutions?
        2. Is the customer satisfied with responses?
        3. Is this a complex technical issue?
        4. Does this require human intervention?
        
        Return escalation decision with reason and priority level.
        """
        
        response = requests.post(
            self.base_url,
            headers={
                "Content-Type": "application/json",
                "Authorization": f"Bearer {self.api_key}"
            },
            json={
                "model": "small-1",
                "stream": False,
                "messages": [{"role": "user", "content": prompt}],
                "functions": self._get_support_tools()
            }
        )
        
        return response.json()
    
    def _get_support_tools(self) -> List[Dict]:
        """Get the customer support tools available"""
        return [
            {
                "name": "search_knowledge_base",
                "description": "Search support knowledge base for solutions",
                "parameters": {
                    "type": "object",
                    "properties": {
                        "query": {"type": "string"},
                        "category": {"type": "string"}
                    },
                    "required": ["query"]
                }
            },
            {
                "name": "check_customer_status",
                "description": "Check customer account status and history",
                "parameters": {
                    "type": "object",
                    "properties": {
                        "customer_id": {"type": "string"},
                        "check_type": {"type": "string", "enum": ["account", "orders", "subscription"]}
                    },
                    "required": ["customer_id"]
                }
            }
        ]
```
